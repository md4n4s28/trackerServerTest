// ===== API CONFIGURATION =====

// 192.168.x.x (your local LAN). You need a publicly reachable address.
//
// Using Railway deployment on port 8080:
//   1. Railway automatically deploys your Node server
//   2. Provides public URL: trackerservertest-production.up.railway.app
//   3. Set API_SERVER_ADDR to "trackerservertest-production.up.railway.app"
//   4. Set API_SERVER_PORT to 8080 (Railway default port)
//   5. Set API_ENDPOINT_PATH to "/api/tracker/telemetry"
//
#define API_SERVER_ADDR     "trackerservertest-production.up.railway.app"
#define API_SERVER_PORT     8080
#define API_ENDPOINT_PATH   "/api/tracker/telemetry"
